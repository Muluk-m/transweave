1. **test-utils.ts** -- Create a test database helper using PGlite in-memory mode:

   ```typescript
   import { PGlite } from '@electric-sql/pglite';
   import { drizzle } from 'drizzle-orm/pglite';
   import { migrate } from 'drizzle-orm/pglite/migrator';
   import * as schema from './schema';
   import type { DrizzleDB } from './drizzle.types';

   export async function createTestDb(): Promise<{ db: DrizzleDB; client: PGlite }> {
     // Use in-memory PGlite for fast, isolated tests
     const client = new PGlite();
     const db = drizzle(client, { schema });

     // Apply migrations
     await migrate(db, { migrationsFolder: './src/db/migrations' });

     return { db: db as DrizzleDB, client };
   }

   export async function cleanupTestDb(client: PGlite): Promise<void> {
     await client.close();
   }
   ```

   NOTE: PGlite with no dataDir argument creates an in-memory database. Each test gets a fresh, isolated database. The `migrate()` function applies the SQL migration files generated by drizzle-kit.

   If `migrate()` does not work with PGlite (open question from research), fall back to `drizzle-kit push` equivalent. Alternative: read the migration SQL file and execute it directly via `client.exec(sql)`.

   FALLBACK test-utils approach if migrate() has issues:
   ```typescript
   import { readFileSync, readdirSync } from 'fs';
   import { join } from 'path';

   export async function createTestDb() {
     const client = new PGlite();
     const db = drizzle(client, { schema });

     // Apply migration SQL files directly
     const migrationsDir = join(__dirname, 'migrations');
     const files = readdirSync(migrationsDir).filter(f => f.endsWith('.sql')).sort();
     for (const file of files) {
       const sql = readFileSync(join(migrationsDir, file), 'utf-8');
       await client.exec(sql);
     }

     return { db: db as DrizzleDB, client };
   }
   ```
